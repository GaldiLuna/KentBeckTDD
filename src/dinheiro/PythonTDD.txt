# SUPER CLASSE
class TestCase(object):
  def __init__(self, name):
      self.name = name

  def run(self, result):
      result.testStarted()
      self.setUp()
      exec "self." + self.name + "()"
      self.tearDown()

  def setUp(self):
      pass

  def tearDown(self):
      pass

# CLASSE PARA A EXECUÇÃO DO TESTE
class WasRun(TestCase):
  def __init__(self, name):
      self.wasRun = None
      TestCase.__init__(self, name)

  def setUp(self):
      self.log = "setUp "

  def tearDown(self):
      self.log = self.log + "tearDown "

#FUNÇÕES PARA EXECUTAR O TESTE
  def testMethod(self):
    self.log = self.log + "testMethod "
